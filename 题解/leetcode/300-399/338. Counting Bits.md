class Solution:
    def countBits(self, n: int) -> List[int]:
        result = [0] * (n + 1)
      
        for i in range(n+1):
            t = i
            cnt = 0
            while t != 0:
                cnt += t % 2
                t = t // 2

            result[i] = cnt

        return result
