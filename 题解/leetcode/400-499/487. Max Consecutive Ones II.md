class Solution:
    def findMaxConsecutiveOnes(self, nums: List[int]) -> int:
        i, j = -1, 0
        n = len(nums)
        result = 0
        count0 = 0
        
        while j < n:
            if count0 <= 1:
                if nums[j] == 0:
                    count0 += 1
                
                j += 1
            else:
                i += 1
                if nums[i] == 0:
                    count0 -= 1
                    
            if count0 <= 1:
                result = max(result, j - i - 1)
                
        return result
