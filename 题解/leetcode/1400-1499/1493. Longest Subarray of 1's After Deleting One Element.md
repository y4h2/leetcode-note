


sliding window
```py
class Solution:
    def longestSubarray(self, nums: List[int]) -> int:
        n = len(nums)
        i, j = 0, 0
        k = 1 # delete one zero
        result = 0
        
        while j < n:
            if nums[j] == 0:
                k -= 1
                
            while k < 0:
                if nums[i] == 0:
                    k += 1
                i += 1
            
            result = max(result, j - i)
            j += 1
            
        return result
```



直觉解法
```py
class Solution:
    def longestSubarray(self, nums: List[int]) -> int:
        n = len(nums)
        arr = [[1, nums[0]]]
        
        for i in range(1,n):
            if nums[i] == arr[-1][1]:
                arr[-1][0] += 1
            else:
                arr.append([1, nums[i]])
                
        if len(arr) == 1 and arr[0][1] == 1:
            return arr[0][0] - 1
        
        
        if arr[0][1] == 0:
            arr.pop(0)
        if len(arr) == 0:
            return 0
        
        result = arr[0][0]
        i = 2
        while i < len(arr):
            if arr[i-1][0] == 1:
                result = max(result, arr[i-2][0] + arr[i][0])
            else:
                result = max(result, arr[i][0])
            i += 2
            
        return result
```
