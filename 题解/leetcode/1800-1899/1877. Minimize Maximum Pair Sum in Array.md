```py
class Solution:
    def minPairSum(self, nums: List[int]) -> int:
        nums.sort()
        n = len(nums)
        i = 0
        result = nums[0] + nums[-1]
        while (i + 1) * 2 <= n:
            result = max(result, nums[i] + nums[-i-1])
            i += 1
            
        return result
```
