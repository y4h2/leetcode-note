2. 对每组的元素之间求解总选择数


求不同group之间两两的pair数

![image.png](assets/image-20220701100212-a7vh4rz.png)



```python
class Solution:
    def countPairs(self, n: int, edges: List[List[int]]) -> int:
        unionFind = UnionFind(n)
        for edge in edges:
            p, q = edge
            unionFind.union(p, q)
          
        unionFind.compress()
        return unionFind.count()
      
class UnionFind:
    def __init__(self, n: int):
        self.parents = [i for i in range(n)]
      
    def find(self, p: int) -> int:
        while p != self.parents[p]:
            self.parents[p] = self.parents[self.parents[p]]
            p = self.parents[p]
          
        return p

    def union(self, p: int, q: int):
        self.parents[self.find(p)] = self.find(q)
      
    def compress(self):
        for i in range(len(self.parents)):
            self.parents[i] = self.find(i)
      
    def count(self) -> int:
        n = len(self.parents)
        values = list(collections.Counter(self.parents).values())
        presum = list(itertools.accumulate(values))
        result = 0
        for i in range(len(values)):
            result += values[i] * (n - presum[i])
        return result
```
