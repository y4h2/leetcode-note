class Solution:
    def numIslands(self, grid: List[List[str]]) -> int:
        m, n = len(grid), len(grid[0])
        
        def dfs(i: int, j: int):
            if not (i >= 0 and i < m and j >= 0 and j < n):
                return
            
            if grid[i][j] == '0':
                return
            
            grid[i][j] = '0'
            
            dfs(i+1, j)
            dfs(i, j+1)
            dfs(i-1, j)
            dfs(i, j-1)
        
        result = 0
        for i in range(m):
            for j in range(n):
                if grid[i][j] == '1':
                    result += 1
                    dfs(i,j)
                    
                    
        return result
