class Solution:
    def numEnclaves(self, grid: List[List[int]]) -> int:
        m, n = len(grid), len(grid[0])
        
        def dfs(i: int, j: int):
            if not (0 <= i and i < m and 0 <= j and j < n):
                return
            if grid[i][j] == 0:
                return

            grid[i][j] = 0

            dfs(i+1, j)
            dfs(i, j+1)
            dfs(i-1, j)
            dfs(i, j-1)
        
            
        for i in range(m):
            dfs(i, 0)
            dfs(i, n-1)
            
        for j in range(n):
            dfs(0, j)
            dfs(m-1, j)
        
        result = 0
        for i in range(m):
            for j in range(n):
                if grid[i][j] == 1:
                    result += 1
                    
        return result
