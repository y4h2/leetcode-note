```py
class Solution:
    def longestOnes(self, nums: List[int], k: int) -> int:
        i = -1
        n = len(nums)
        result = 0
        count0 = 0
        
        for j in range(n):
            if nums[j] == 0:
                count0 += 1
            
            while i < j and count0 > k:
                i += 1
                if nums[i] == 0:
                    count0 -= 1
                    
            result = max(result, j - i)
            
        return result
```

```py
class Solution:
    def longestOnes(self, nums: List[int], k: int) -> int:
        i, j = -1, 0
        n = len(nums)
        result = 0
        count0 = 0
        
        while j < n:
            if count0 <= k:
                if nums[j] == 0:
                    count0 += 1
                
                j += 1
            else:
                i += 1
                if nums[i] == 0:
                    count0 -= 1
                    
            if count0 <= k:
                result = max(result, j - i - 1)
                
        return result
```
