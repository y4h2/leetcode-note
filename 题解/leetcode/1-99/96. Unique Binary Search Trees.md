class Solution:
    @lru_cache
    def numTrees(self, n: int) -> int:
        if n == 0:
            return 1
        if n == 1:
            return 1
        if n == 2:
            return 2
      
        result = 0
        for i in range(n):
            result += self.numTrees(n - 1 - i) * self.numTrees(i)
              
        return result
