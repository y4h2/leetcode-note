层次遍历


注意这种情况

![image](assets/image-20220723014351-uq8iy2e.png)


```python
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def isCompleteTree(self, root: Optional[TreeNode]) -> bool:
        if not root:
            return True
        queue = deque([])
        queue.append(root)
      
        while len(queue) > 0:
            cur = queue.popleft()
            if not cur:
                break
            queue.append(cur.left)
            queue.append(cur.right)
      
        while len(queue) > 0:
            cur = queue.pop()
            if cur != None:
                return False
      
        return True
          
```
