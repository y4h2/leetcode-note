class Solution:
    def makeFancyString(self, s: str) -> str:
        if len(s) <= 2:
            return s
        
        result = ""
        n = len(s)
        result = s[:2]
        
        for i in range(2, n):
            if s[i] == s[i-1] and s[i-1] == s[i-2]:
                continue
            
            result += s[i]
            
        return result
