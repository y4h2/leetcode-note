class Solution:
    def minMaxGame(self, nums: List[int]) -> int:
        if len(nums) == 1:
            return nums[0]
        n = len(nums)
        newNums = [0] * (n // 2)
        for i in range(n//2):
            if i % 2 == 0:
                newNums[i] = min(nums[i*2], nums[i*2+1])
            else:
                newNums[i] = max(nums[i*2], nums[i*2+1])
      
        return self.minMaxGame(newNums[:])
