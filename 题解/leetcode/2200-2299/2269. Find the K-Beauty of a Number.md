```python
class Solution:
    def divisorSubstrings(self, num: int, k: int) -> int:
        result = 0
        s = str(num)
        n = len(s)
        for i in range(n-k+1):
            subnum = int(s[i:i+k])
            if subnum == 0:
                continue
            if num % subnum == 0:
                result += 1
              
        return result
```
