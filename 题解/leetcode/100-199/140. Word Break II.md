class Solution:
    def wordBreak(self, s: str, wordDict: List[str]) -> List[str]:
        self.trie = Trie()
        for word in wordDict:
            self.trie.add(word)
            
        self.result = []
        self.dfs(s, [])
        
        return self.result
            
    def dfs(self, s: str, cur: List[str]):
        if s == "":
            self.result.append(' '.join(cur))
        for prefixPos in self.trie.findPrefix(s):
            self.dfs(s[prefixPos+1:], cur[:] + [s[:prefixPos+1]])
        
        
class TrieNode:
    def __init__(self, c: str):
        self.value = c
        self.children = {}
        self.isWord = False
        
class Trie:
    def __init__(self):
        self.root = TrieNode("")
        self.memo = set() # only record true result; false might become true after
        
    def add(self, word: str):
        cur = self.root
        for c in word:
            if c not in cur.children:
                cur.children[c] = TrieNode(c)
            
            cur = cur.children[c]
        
        
        cur.isWord = True
    
    def findPrefix(self, word: str) -> List[int]:
        result = []

        cur = self.root
        for i, c in enumerate(word):
            if c not in cur.children:
                return result
            # 注意cur的变换位置
            cur = cur.children[c]
            if cur.isWord:
                result.append(i)

        return result
